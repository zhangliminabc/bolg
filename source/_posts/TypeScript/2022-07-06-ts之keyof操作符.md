### keyof: 用于获取某种类型的所有键，其返回类型是联合类型

```ts

interface Person {
	name: string;
	age: number;
	location:string;
}

type K1 = keyof Person; // name || age || location
type K2 = keyof Person[] // name || age || location || push || concat ................
type K3 = keyof { [x: string]: Person } // string || number


type Todo = {
	id: number;
	text: string;
	done: boolean;
}

const todo: Todo = {
	id: 1, 
	text: 'Learn TypeScript keyof',
	done: false
}

function prop<T extends object, K extends keyof T>(obj: T, key: K): T[K] {
	return obj[key]
}
```
